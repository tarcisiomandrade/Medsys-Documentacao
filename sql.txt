create database medsys;
use medsys;

create table paciente(
pk cod int identity(1,1),
nome varchar(50) not null,
cpf varchar(50) unique,
rua varchar(50) not null,

primary key(matricula)
);


create table consulta(
pk codigo int identity(1,1),
valor decimal,
data datetime not null,
fk paciente int,
fk medico int,
fk funcionario int

primary key(cod)
foreign key paciente
  references paciente(cod),
foreign key medico
  references medico(cod),
foreign key funcionario
  references funcionario(cod),
);


create table medico(
pk cod int identity(1,1),
nome varchar(50),
cpf varchar(50) unique,
datanasc datetime,

primary key(cod)
);


create table funcionario(
pk cod int identity(1,1),
nome varchar(50) not null,
cpf varchar(50) unique,
datanasc datetime,

primary key(cod)
);


create table servico(
pk cod int identity(1,1),
valor decimal not null,
descricao varchar(100) not null,

primary key(cod)
);


create table tratamento(
pk cod int identity(1,1),
valor decimal not null,
data datetime not null,
fk medico int,
fk paciente int,
fk consulta int,
fk servico int,

primary key(cod)
foreign key medico
  references medico(cod),
foreign key paciente
  references paciente(cod),
foreign key consulta
  references consulta(cod),
foreign key servico
  references servico(cod),
);

create table usuario(
pk cod int identity(1,1),
nome varchar(100)not null,
responsavel varchar(100),

primary key(cod)
);

create table endereco(
pk cod int identity(1,1),
rua varchar(50),
numero varchar(50) ,
rua varchar(50),
bairro varchar(50),
cidade varchar(50),
estado varchar(50),
numero int,

primary key(cod)
);

creat table telefone(
pk cod identity (1,1),
numero varchar (50)

primary key (cod)
);

Views

Listagem dos pacientes

create view pacientes
as

	select p.nome, p.cpf, p.datanasc
	from paciente as p
	
	order by p.nome


Listagem dos medicos

create view medicos
as

	select m.nome, m.cpf, m.datanasc
	from medico as m
	join paciente as p on p.cod = m.paciente
	
	order by p.nome


Listagem dos funcionarios

create view funcionarios
as

	select f.nome, f.cpf, f.datanasc
	from funcionario as f
	join tratamento as t on t.cod = f.tratamento
	
	order by p.nome
